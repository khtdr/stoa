   (var a 0)
   (var temp)
   (block 
      (var b 1)
      (while (< a 5000) 
         (block 
            (block 
               (print a)
               (= temp a)
               (= a b)

            (= b (+ temp b))
         )
      )
   )
   (fun fib (let [n] (block 
      (if (<= n 1) (return n))

   )))
